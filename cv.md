function sortByVowels(...words) {
    function countVowels(word) {
        const vowels = ['a', 'e', 'i', 'o', 'u'];
        let count = 0;
        for (let i = 0; i < word.length; i++) {
            if (vowels.includes(word[i].toLowerCase())) {
                count++;
            }
        }
        return count;
    }

    return words.sort((a, b) => countVowels(a) - countVowels(b)).join(' ');
}
